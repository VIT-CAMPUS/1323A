#include <stdio.h>

void lruPageReplacement(int pages[], int n, int capacity) {
    int frames[capacity], counter[capacity], time = 0, pageFaults = 0;
    for (int i = 0; i < capacity; i++) {
        frames[i] = -1;
        counter[i] = 0;
    }

    for (int i = 0; i < n; i++) {
        int found = 0, lru = 0;

        for (int j = 0; j < capacity; j++) {
            if (frames[j] == pages[i]) {
                found = 1;
                counter[j] = ++time;
                break;
            }
        }

        if (!found) {
            for (int j = 0; j < capacity; j++) {
                if (frames[j] == -1) {
                    lru = j;
                    break;
                } else if (counter[j] < counter[lru]) {
                    lru = j;
                }
            }

            frames[lru] = pages[i];
            counter[lru] = ++time;
            pageFaults++;
        }

        printf("Step %d: ", i + 1);
        for (int j = 0; j < capacity; j++) {
            if (frames[j] != -1)
                printf("%d ", frames[j]);
            else
                printf("- ");
        }
        printf("\n");
    }

    printf("Total Page Faults: %d\n", pageFaults);
}

int main() {
    int n, capacity;

    printf("Enter the number of pages: ");
    scanf("%d", &n);
    int pages[n];

    printf("Enter the reference string: ");
    for (int i = 0; i < n; i++) scanf("%d", &pages[i]);

    printf("Enter the frame capacity: ");
    scanf("%d", &capacity);

    lruPageReplacement(pages, n, capacity);
    return 0;
}
